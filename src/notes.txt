    function stackBlock( arr, col ) {
        let maxHeightIndex = arr.length;
        let currentHighestBlock = -1;
        for( let i = 0; i < arr.length; i++ ) {
            if( arr[i][col] == 0  ) {
                currentHighestBlock = i;
                break;
            }
        }
        if( ( currentHighestBlock == maxHeightIndex ) || ( currentHighestBlock == -1 ) ) {
            return false
        }
        arr[currentHighestBlock][col] = 1
    }
